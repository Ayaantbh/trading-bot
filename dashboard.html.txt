        // FIX 10 & 11: Helper for robust fetch with timeout and retry for smoother, non-internet-dependent UI
        async function robustFetch(url, options = {}, retries = 2) {
            for (let i = 0; i <= retries; i++) {
                try {
                    const controller = new AbortController();
                    const timeoutId = setTimeout(() => controller.abort(), 8000); // 8s timeout to avoid stuck loading

                    const response = await fetch(url, {
                        ...options,
                        signal: controller.signal
                    });

                    clearTimeout(timeoutId);

                    if (response.ok) {
                        return await response.json();
                    } else if (i === retries) {
                        throw new Error(`HTTP ${response.status}`);
                    }
                } catch (e) {
                    if (i === retries) {
                        console.error(`Fetch failed after ${retries} retries:`, e);
                        // Fallback for offline: return cached or default data
                        if (url.includes('/api/performance')) return { current_equity: 10000, available_cash: 10000, win_rate: 0, daily_pnl: 0, market_status: '🔴 OFFLINE' };
                        if (url.includes('/api/symbols')) return watchlistData; // Use last known
                        if (url.includes('/news')) return {}; // Empty news offline
                        throw e;
                    }
                    await new Promise(resolve => setTimeout(resolve, 1000 * (i + 1))); // Exponential backoff
                }
            }
        }

        async function syncWatchlist() {

                watchlistData = await robustFetch('/api/symbols');
                watchlist = watchlistData.map(item => item.symbol);
                updateWatchlist();

                    updateDashboard().catch(e => console.log('Dashboard init:', e))

                await Promise.race([

        async function loadChart(symbol = symbolSelect.value, containerId = 'chart') {
            const btn = document.getElementById('loadChartBtn');
            btn.disabled = true;
            btn.textContent = 'Loading...';
            const container = document.getElementById(containerId);
            container.innerHTML = '<div class="loading-placeholder">Loading chart data...</div>';
            try {
                const data = await robustFetch(`/api/candlestick/${symbol}`);
                
                if (data.error) {
                    container.innerHTML = `<div class="error-message">❌ ${data.error}</div>`;
                    return;
                }
                
                if (data.candles && data.candles.length) {
                    container.innerHTML = '';
                    const chartInstance = LightweightCharts.createChart(container, {
                        width: container.clientWidth,
                        height: containerId === 'fullscreenChart' ? container.clientHeight : 350,
                        layout: { 
                            background: { color: 'rgba(10,10,20,0.95)' }, 
                            textColor: '#e2e8f0' 
                        },
                        grid: { 
                            vertLines: { color: 'rgba(120,120,180,0.2)' }, 
                            horzLines: { color: 'rgba(120,120,180,0.2)' } 
                        },
                        timeScale: { 
                            borderColor: 'rgba(120,120,180,0.4)',
                            timeVisible: true
                        },
                        rightPriceScale: { 
                            borderColor: 'rgba(120,120,180,0.4)' 
                        }
                    });
                    
                    const candlestickSeries = chartInstance.addCandlestickSeries({
                        upColor: '#22c55e', 
                        downColor: '#ef4444', 
                        borderVisible: false,
                        wickUpColor: '#22c55e', 
                        wickDownColor: '#ef4444'
                    });
                    
                    candlestickSeries.setData(data.candles);
                    chartInstance.timeScale().fitContent();
                    
                    if (containerId === 'chart') {
                        chart = chartInstance;
                    } else {
                        fullscreenChart = chartInstance;
                    }
                    
                    lastPrices[symbol] = data.candles[data.candles.length-1].close;
                }
            } catch(e) {
                // FIX 7 & 10: Timeout/error handling for stuck chart loading
                container.innerHTML = '<div class="error-message">Chart load timed out. Please retry or check connection.</div>';
                console.error('Chart error:', e);
            } finally {
                btn.disabled = false;
                btn.textContent = '🔄 Load Chart';
            }
        }

        function updateNews(newsData) {
            const container = document.getElementById('newsFeed');
            let html = '';
            let hasNews = false;
            
            for (const [symbol, articles] of Object.entries(newsData || {})) {
                if(articles && articles.length > 0) {
                    hasNews = true;
                    html += `<div style="margin-bottom: 1.5rem;">
                        <h4 style="margin-bottom: 0.8rem;color:#8b5cf6; display: flex; align-items: center; gap: 0.5rem;">
                            📰 ${symbol}
                        </h4>`;
                    articles.slice(0,3).forEach(article => {
                        const colorClass = article.color === 'green' ? 'bull' : article.color === 'red' ? 'bear' : 'neutral';
                        const sentimentEmoji = article.sentiment > 0 ? '🟢' : article.sentiment < 0 ? '🔴' : '⚪';
                        html += `<div class="news-item ${colorClass}">
                            <div style="font-weight: 600; margin-bottom: 0.4rem; font-size: 0.95rem;">${article.headline}</div>
                            <div style="font-size: 0.85rem; opacity: 0.8; display: flex; justify-content: space-between;">
                                <span>${article.time}</span>
                                <span>${sentimentEmoji}</span>
                            </div>
                        </div>`;
                    });
                    html += '</div>';
                }
            }
            
            if (!hasNews) {
                html = '<div style="color:var(--text-secondary); text-align: center; padding: 2rem;">No news available for tracked symbols</div>';
            }
            
            container.innerHTML = html;
        }

        async function updateDashboard() {

                const perf = await robustFetch('/api/performance');

                document.getElementById('winRate').textContent = (perf.win_rate || 0).toFixed(1);

                document.getElementById('availableCash').textContent = perf.available_cash?.toLocaleString('en-US', {minimumFractionDigits: 2, maximumFractionDigits: 2}) || '0.00';
                
                document.getElementById('statusDiv').textContent = perf.market_status || 'Loading...';
                document.getElementById('marketStatus').textContent = perf.market_status || 'Loading...';
                
                // Update button states
                pauseState = perf.trading_paused;
                document.getElementById('pauseBtn').textContent = pauseState ? '▶️ Resume Trading' : '⏸️ Pause Trading';
                
                autoTradingState = perf.auto_trading;
                const toggleBtn = document.getElementById('toggleTradingBtn');
                if (toggleBtn) {
                    toggleBtn.textContent = autoTradingState ? '⏹️ Stop Auto Trading' : '🔄 Start Auto Trading';
                }

                const positions = await robustFetch('/api/positions');
                updatePositions(positions);
            } catch(e) {
                // FIX 7 & 10: Graceful degradation for stuck loading
                console.error('Dashboard update failed:', e);
                document.getElementById('statusDiv').textContent = '⚠️ Connection Error - Using cached data';
            }
        }
        
        function updatePositions(positions) {
            const tbody = document.querySelector('#positionsTable tbody');
            tbody.innerHTML = '';
            
            let hasPositions = false;
            
            for (const [symbol, info] of Object.entries(positions || {})) {
                if (info.shares > 0) {
                    hasPositions = true;
                    const currentPrice = lastPrices[symbol] || info.entry;
                    const pnl = ((currentPrice - info.entry) / info.entry * 100) || 0;
                    const pnlClass = pnl >= 0 ? 'profit' : 'loss';
                    tbody.innerHTML += `
                        <tr>
                            <td><strong>${symbol}</strong></td>
                            <td>${Math.abs(info.shares).toLocaleString()}</td>
                            <td>$${info.entry?.toFixed(2) || '0.00'}</td>
                            <td class="${pnlClass}">${pnl.toFixed(2)}%</td>
                            <td>$${info.stop?.toFixed(2) || '0.00'}</td>
                            <td>$${info.tp?.toFixed(2) || '0.00'}</td>
                        </tr>`;
                }
            }
            
            if (!hasPositions) {
                tbody.innerHTML = `
                    <tr>
                        <td colspan="6" style="text-align: center; padding: 2rem; color: var(--text-secondary);">
                            No open positions
                        </td>
                    </tr>`;
            }
        }
        
        function setupControls() {
            const controls = [
                { range: 'stop_loss_pct', val: 'stop_loss_pct_val' },
                { range: 'take_profit_pct', val: 'take_profit_pct_val' },
                { range: 'rsi_low', val: 'rsi_low_val' },
                { range: 'rsi_high', val: 'rsi_high_val' },
                { range: 'ma_short_period', val: 'ma_short_val' },
                { range: 'ma_long_period', val: 'ma_long_val' },
                { range: 'volume_multiplier', val: 'volume_mult_val' },
                { range: 'max_open_risk_pct', val: 'max_risk_val' },
                { range: 'position_size_pct', val: 'pos_size_val' }
            ];

            controls.forEach(({ range, val }) => {
                const rangeEl = document.getElementById(range);
                const valEl = document.getElementById(val);
        
                if (rangeEl && valEl) {
                    rangeEl.oninput = () => {
                        valEl.value = rangeEl.value;
                    };
            
                    valEl.oninput = () => {
                        const min = parseFloat(rangeEl.min);
                        const max = parseFloat(rangeEl.max);
                        const value = parseFloat(valEl.value);
                        
                        if (value >= min && value <= max) {
                            rangeEl.value = valEl.value;
                        } else {
                            valEl.value = rangeEl.value;
                        }
                    };
                }
            });
        }

            document.getElementById('logs').innerHTML = 'Logs cleared.';

        // SSE logs
        if (typeof(EventSource) !== 'undefined') {
            try {
                const source = new EventSource('/logs');
                source.addEventListener('message', e => {
                    try {
                        const data = JSON.parse(e.data);
                        if (data.type === 'log_update') {
                            const logs = document.getElementById('logs');
                            const entry = document.createElement('div');
                            entry.className = `log-entry ${data.level || ''}`;
                            entry.textContent = data.log;
                            logs.appendChild(entry);
                            logs.scrollTop = logs.scrollHeight;
                        }
                    } catch(err) {}
                });
                
                source.onerror = (e) => {
                    console.log('SSE connection error, retrying...');
                    setTimeout(() => {
                        if (source.readyState === EventSource.CLOSED) {
                            // Reconnect
                            location.reload();
                        }
                    }, 5000);
                };
            } catch(e) {
                document.getElementById('logs').innerHTML = 'Live logs unavailable';
            }
        }
